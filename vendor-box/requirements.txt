# Pragmatic install constraints for the vendor box.
#
# We use lower bounds (>=) instead of exact pins because pinning to a single
# version forces pip to find that exact wheel for your Python + OS + arch,
# which often fails. Lower bounds let pip pick a compatible set.
#
# Tested combination on RTX 4080 / Ubuntu 22.04 / Python 3.10:
#   torch 2.5.1 + cu121, numpy 1.26.4, diffusers 0.31.0
#
# DO NOT install on Mac — torch+cu121 wheels are Linux/Windows x86_64 only.

# ── Web service ────────────────────────────────────────────────────────────
fastapi>=0.110,<1
uvicorn[standard]>=0.30
pydantic>=2.6,<3
httpx>=0.27
python-multipart>=0.0.9

# ── Image / video / numerics ──────────────────────────────────────────────
pillow>=10.0
# opencv 4.13+ bumped to numpy>=2; we keep numpy<2 for the rest of the stack
# (mediapipe ABI, DM-VTON kernel modules). Cap opencv accordingly.
opencv-python-headless>=4.9,<4.13
# Keep numpy < 2 — many AI libs (mediapipe, parts of diffusers) still bundle
# extensions compiled against numpy 1.x ABI. Upgrading to 2.x causes
# "module compiled against numpy 1.x" runtime errors that are hard to debug.
numpy>=1.24,<2

# ── Torch — CUDA 12.1 wheels ──────────────────────────────────────────────
# Lower bound only; pip will pick the latest cu121 wheel that matches your Python.
--extra-index-url https://download.pytorch.org/whl/cu121
torch>=2.4,<3
torchvision>=0.19,<1

# ── Diffusers + transformers stack ────────────────────────────────────────
diffusers>=0.30
transformers>=4.45
accelerate>=1.0
safetensors>=0.4
einops>=0.7

# ── Body parsing (for VTON pre-processing) ────────────────────────────────
# Any modern 0.10.x or 0.11.x works — dm_vton_loader uses the Tasks API
# (`mediapipe.tasks.python.vision.PoseLandmarker`), which is the supported
# replacement for the removed legacy `mediapipe.solutions.pose` namespace.
mediapipe>=0.10.14

# ── HuggingFace model downloads ───────────────────────────────────────────
huggingface_hub>=0.25
